def standardize_column_names(df):
    """
    Standardize column names by making them lowercase and replacing spaces with underscores.
    """
    df.columns = df.columns.str.lower().str.replace(" ", "_")
    return df

def coerce_numeric_and_handle_missing(df):
    """
    Convert numeric columns and handle missing values by median imputation.
    """
    for col in df.select_dtypes(include=["object"]).columns:
        try:
            df[col] = pd.to_numeric(df[col])
        except ValueError:
            pass

    for col in df.select_dtypes(include=["int64", "float64"]).columns:
        df[col] = df[col].fillna(df[col].median())

    return df

def plot_normalized_performance_over_time(df, date_col="date", freq="D"):
    """
    Plot normalized ROI, conversion rate, and acquisition cost over time.
    
    Steps:
    - Parse and sort by date
    - Aggregate to a chosen frequency (default daily)
    - Min-max normalize metrics for comparability
    """
    cols = ["roi", "conversion_rate", "acquisition_cost"]

    # Work on a copy with required columns
    data = df[[date_col] + cols].copy()

    # Parse date
    data[date_col] = pd.to_datetime(data[date_col], errors="coerce")
    data = data.dropna(subset=[date_col])

    # Coerce metrics to numeric (handles %, $, commas)
    for c in cols:
        data[c] = (
            data[c]
            .astype(str)
            .str.replace('%', '', regex=False)
            .str.replace('$', '', regex=False)
            .str.replace(',', '', regex=False)
            .str.strip()
        )
        data[c] = pd.to_numeric(data[c], errors="coerce")

    # Fill missing numeric values with median (EDA-friendly)
    data[cols] = data[cols].fillna(data[cols].median(numeric_only=True))

    # Sort and set index for resampling
    data = data.sort_values(date_col).set_index(date_col)

    # Aggregate over time (D=daily, W=weekly, M=monthly)
    agg = data[cols].resample(freq).mean()

    # Min-max normalize (avoid div by zero)
    denom = (agg.max() - agg.min()).replace(0, np.nan)
    norm = (agg - agg.min()) / denom
    norm = norm.fillna(0)

    plt.figure(figsize=(10, 5))
    plt.plot(norm.index, norm["roi"], label="ROI", linewidth=2)
    plt.plot(norm.index, norm["conversion_rate"], label="Conversion Rate", linewidth=2)
    plt.plot(norm.index, norm["acquisition_cost"], label="Acquisition Cost", linewidth=2)

    plt.title(f"Normalized Marketing Performance Metrics Over Time (resample={freq})")
    plt.xlabel("Date")
    plt.ylabel("Normalized Value (01)")
    plt.legend()
    plt.tight_layout()
    plt.show()

def plot_performance_heatmap_normalized(df, group_col):
    """
    Plot a normalized heatmap of average performance metrics grouped
    by a selected categorical variable.

    Normalization is applied column-wise (min-max scaling) so that
    metrics with different units can be compared visually.
    """
    metrics = ['conversion_rate', 'acquisition_cost', 'roi', 'clicks']

    data = df[[group_col] + metrics].copy()

    # Coerce metrics to numeric
    for c in metrics:
        data[c] = (
            data[c]
            .astype(str)
            .str.replace('%', '', regex=False)
            .str.replace('$', '', regex=False)
            .str.replace(',', '', regex=False)
            .str.strip()
        )
        data[c] = pd.to_numeric(data[c], errors='coerce')

    # Drop rows with missing group
    data = data.dropna(subset=[group_col])

    # Fill missing metric values with median
    data[metrics] = data[metrics].fillna(
        data[metrics].median(numeric_only=True)
    )

    # Group by category and compute mean
    grouped = data.groupby(group_col)[metrics].mean()

    # Min-max normalize *per metric*
    grouped_norm = (grouped - grouped.min()) / (grouped.max() - grouped.min())
    grouped_norm = grouped_norm.fillna(0)

    plt.figure(figsize=(10, 6))
    sns.heatmap(
        grouped_norm,
        cmap="coolwarm",
        annot=True,
        fmt=".2f",
        cbar_kws={"label": "Normalized Value (01)"}
    )

    plt.title(
        f"Normalized Performance Metrics by {group_col.replace('_', ' ').title()}"
    )
    plt.xlabel("Metrics")
    plt.ylabel(group_col.replace('_', ' ').title())
    plt.tight_layout()
    plt.show()

Summary and Interpretation

The dataset shows that marketing performance cannot be understood through spend alone. While acquisition costs remain relatively stable across campaigns and over time, efficiency metrics such as conversion rate and ROI vary more significantly. This indicates that how marketing resources are allocated is more influential than the total amount spent.

Normalized visualizations revealed subtle but meaningful differences between campaign types that were not visible in raw averages. In particular, some campaign types consistently demonstrate higher relative ROI and conversion efficiency despite similar acquisition costs (ex: Search, Influencer). Additionally, the time-series analysis shows that short-term fluctuations in ROI and conversion rate occur independently of spending levels.

This analysis assumes consistent metric definitions across all campaigns and does not account for external factors such as seasonality, audience overlap, or channel-specific attribution effects. The lack of detailed temporal context limits causal interpretation, and normalization may smooth out extreme values that could be operationally significant.

A surprising finding was the similarity of raw performance averages across campaign types, which initially suggested minimal differentiation. However, normalized analysis revealed relative performance differences, highlighting the importance of scale-aware exploratory techniques. Without normalization, these insights would have remained hidden.
